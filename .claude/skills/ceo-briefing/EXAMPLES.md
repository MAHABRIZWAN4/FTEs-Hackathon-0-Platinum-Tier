## Programmatic Usage

```python
from scripts.ceo_briefing import generate_ceo_briefing
from datetime import datetime

# Generate for current week
report_path = generate_ceo_briefing()
print(f"Report: {report_path}")

# Generate for specific week
specific_date = datetime(2026, 2, 24)
report_path = generate_ceo_briefing(specific_date)

# Read the report
with open(report_path, 'r') as f:
    briefing = f.read()
    print(briefing)
```

## Get Specific Data

```python
from scripts.ceo_briefing import (
    get_completed_tasks,
    get_pending_tasks,
    get_financial_summary,
    get_system_health,
    get_week_range
)

# Get week range
week_start, week_end = get_week_range()
print(f"Week: {week_start} to {week_end}")

# Get completed tasks
completed = get_completed_tasks(week_start, week_end)
print(f"Completed: {len(completed)} tasks")

# Get financial summary
financial = get_financial_summary()
print(f"Net Profit: ${financial['net_profit']:,.2f}")

# Get system health
health = get_system_health()
print(f"Success Rate: {health['success_rate']:.1f}%")
```

## Integration with Email

```python
from scripts.ceo_briefing import generate_ceo_briefing
from scripts.send_email import send_email

# Generate briefing
report_path = generate_ceo_briefing()

# Read report
with open(report_path, 'r') as f:
    briefing_content = f.read()

# Send via email
send_email(
    to="ceo@company.com",
    subject="Weekly CEO Briefing - Week of March 2, 2026",
    body=briefing_content,
    html=False
)
```
